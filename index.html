<script>  
    // Initialize Icons  
    lucide.createIcons();  

    function checkInputs() {  
        // Optional: Enable/Disable button based on input validity  
    }  

    function calculate() {  
        // Get inputs  
        const lengthInput = document.getElementById("length");  
        const tempInput = document.getElementById("temperature");  
          
        const L = parseFloat(lengthInput.value);  
        const T = parseFloat(tempInput.value);  

        // Validation  
        if (isNaN(L) || isNaN(T)) {  
            alert("Please enter valid numbers for both Length and Temperature.");  
            return;  
        }  

        // Calculation Logic (From user snippet)  
        // valX = 1300 + (L * 0.000017 * (35 - T) * 1000);  
        // The user had * 1000 inside the parenthesis.   
        // 0.000017 is likely coefficient. 1000 converts m to mm.  
          
        const expansionFactor = L * 0.000017 * (35 - T) * 1000;  
          
        const valX = 1300 + expansionFactor;  
        const valY = 2300 + 3 * expansionFactor;  

        // Update DOM  
        document.getElementById("resX_mm").textContent = valX.toFixed(2);  
        document.getElementById("resX_m").textContent = (valX / 1000).toFixed(4);  
          
        document.getElementById("resY_mm").textContent = valY.toFixed(2);  
        document.getElementById("resY_m").textContent = (valY / 1000).toFixed(4);  

        // Show results with animation  
        const resultsDiv = document.getElementById("results");  
        resultsDiv.classList.remove("hidden");  
          
        // Smooth scroll to results on small screens  
        if(window.innerWidth < 768) {  
            setTimeout(() => {  
                resultsDiv.scrollIntoView({ behavior: 'smooth', block: 'nearest' });  
            }, 100);  
        }  
    }  

    function clearForm() {  
        document.getElementById("length").value = "";  
        document.getElementById("temperature").value = "";  
        document.getElementById("results").classList.add("hidden");  
    }  
</script>
